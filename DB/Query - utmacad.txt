	marital_stat	nchar(10),
	religion		nchar(10),
	race			nvarchar(20),
	nationality		nvarchar(50),
	student_address	text,
	designation		nvarchar(50),
	course			nvarchar(100),
	faculty			nvarchar(100),
	semester		int,
	session_no		nvarchar(10),
	category		nchar(10),
	branch			nchar(10),
	remarks			text,
